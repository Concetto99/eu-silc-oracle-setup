-- GROUP BY GEOGRAPHIC AREA
select DISTINCT DB040 AS AREA, COUNT(*) from C##EUSILC_SVIL.TD_UDB_cIT23D
GROUP BY DB040;

-- TOTAL NUMBER OF MALE AND FEMALE IN THE NORTH-OVEST AREA OF ITALY (ITC)
SELECT DISTINCT PB150 AS SEX, COUNT(*) AS TOTAL FROM C##EUSILC_SVIL.TD_UDB_cIT23P
WHERE PX030 IN (
                select DB030 from C##EUSILC_SVIL.TD_UDB_cIT23D
                where DB040 = 'ITC'
                )
GROUP BY PB150;
